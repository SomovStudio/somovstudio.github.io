События EventHandler

Применение делегатов EventHandler<TEventArgs> и EventHandler


class MyEvent 
{
    public event EventHandler SomeEvent; // делегат EventHandler
    
    public void OnSomeEvent()
    {
        if(SomeEvent != null) SomeEvent(this, EventArgs.Empty);
    }
}

static class EventDemo
{
    static void Handler(object source, EventArgs arg)
    {
        Console.WriteLine("Произошло событие. Источник = " + source + " | Аргумент = " + arg);
    }
    
    public static void test()
    {
        MyEvent evt = new MyEvent();
        evt.SomeEvent += Handler;
        evt.OnSomeEvent();
    }
}

Практический пример обработки событий


class KeyEventArgs: EventArgs
{
    public char ch;
}

class KeyEvent
{
    public event EventHandler<KeyEventArgs> KeyPress;
    
    public void OnKeyPress(char key)
    {
        KeyEventArgs k = new KeyEventArgs();
        if(KeyPress != null)
        {
            k.ch = key;
            KeyPress(this, k);
        }
    }
}

static class KeyEventDemo
{
    public static void Test()
    {
        KeyEvent kevt = new KeyEvent();
        ConsoleKeyInfo key;
        int count = 0;
        
        kevt.KeyPress += (sender, e) => Console.WriteLine("Событие: Нажата кнопка " + e.ch);
        
        kevt.KeyPress += (sender, e) => count++;
        
        Console.WriteLine("Введите несколько символов. В конце поставьте точку.");
        do {
            key = Console.ReadKey();
            kevt.OnKeyPress(key.KeyChar);
        } while(key.KeyChar != '.');
        
        Console.WriteLine("Было нажато " + count + " клавиш.");
    }
}
