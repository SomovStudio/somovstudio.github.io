Индексаторы

Создание одномерного индексатора


public class TestArray1
{
    private int[] a;
    public int Length;
    
    public TestArray1(int size)
    {
        a = new int[size];
        Length = a.Length;
    }
    
    // Индексатор для класса
    public int this[int index]
    {
        // Аксессор get
        get {
            if(index < Length && index >= 0) return a[index];
            return 0;
        }
        // Аксессор set
        set {
            if(index < Length && index >= 0) a[index] = value;
        }
    }
}


Пример использования


TestArray1 test1 = new TestArray1(5);
for(int i = 0; i < (test1.Length * 2); i++)
    test1[i] = i + 10;

for(int j = 0; j < (test1.Length * 2); j++)
    Console.WriteLine(test1[j]);

Перегрузка индексаторов


public class TestArray1
{
    private int[] a;
    public int Length;
    
    public TestArray1(int size)
    {
        a = new int[size];
        Length = a.Length;
    }
    
    // Индексатор для класса
    public int this[int index]
    {
        // Аксессор get
        get {
            if(index < Length && index >= 0) return a[index];
            return 0;
        }
        // Аксессор set
        set {
            if(index < Length && index >= 0) a[index] = value;
        }
    }
    
    // Перегруженный индексатор для класса
    public int this[double indexD]
    {
        // Аксессор get
        get {
            int index = (int) indexD;
            if(index < Length && index >= 0) return a[index];
            return 0;
        }
        // Аксессор set
        set {
            int index = (int) indexD;
            if(index < Length && index >= 0) a[index] = value;
        }
    }
}



Пример использования


TestArray1 test1 = new TestArray1(5);
test1[1.2] = 1200;
Console.WriteLine("Индекс 1.2: " + test1[1.2]);

Многомерные индексаторы


public class TestArray2
{
    private int[,] a;
    private int rows, cols;
    public int Length;
    
    public TestArray2(int r, int c)
    {
        rows = r;
        cols = c;
        a = new int[rows, cols];
        Length = rows * cols;
    }
    
    // Индексатор для класса
    public int this[int indexR, int indexC]
    {
        // Аксессор get
        get {
            return a[indexR, indexC];
        }
        // Аксессор set
        set {
            a[indexR, indexC] = value;
        }
    }
}


Пример использования


TestArray2 test2 = new TestArray2(3, 5);
test2[0, 1] = 550;
Console.WriteLine("Индекс 0, 1: " + test2[0, 1]);
