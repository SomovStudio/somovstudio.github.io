Анонимные методы.


delegate void CountIt();

public static class AnonMethDemo 
{
    public static void Test()
    {
        CountIt count = delegate {
            for(int i=0; i < 5; i++)
                Console.WriteLine("Анонимный метод. Счетчик = " + i);
        };
        
        count();
    }
}

Возврат значения из анонимного метода


delegate int CountIt(int end);

public static class AnonMethDemo 
{
    public static void Test()
    {
        CountIt count = delegate (int end) {
            int sum = 0;
            for(int i=0; i < end; i++)
            {
                sum += i;
            }
            return sum;
        };
        
        int result = count(5);
        Console.WriteLine("Анонимный метод вернул значение: " + result);                    
    }
}




Применение внешних переменных в анонимных методах


delegate int CountIt(int end);

public static class AnonMethDemo 
{
    static CountIt Counter() 
    {
        int sum = 0;
        CountIt ctObj = delegate (int end) 
        {
            for(int i=0; i <= end; i++) {
                sum += i;
            }
            return sum;
        };
        return ctObj;
    }

    public static void Test()
    {
        CountIt count = Counter();        
        int result = count(3);
        Console.WriteLine("Анонимный метод вернул значение: " + result);                    
    }
}
