SKIP - пропускает указанное количество элементов


string[] users = {"John", "Jackson", "Bill", "Admin", "User", "sa"};
IEnumerable<string> values = users.Skip(2);
foreach(var value in values)
{
    Console.WriteLine(value);
}

SKIP WHILE - пропускает элементы до тех пор пока условие истино


string[] users = {"John", "Jackson", "Bill", "Admin", "User", "sa"};
//IEnumerable<string> values = users.SkipWhile(u => u.StartsWith("J"));
IEnumerable<string> values = users.SkipWhile( (s, i) => s.Length < 10 && i < 3 );
foreach(var value in values)
{
    Console.WriteLine(value);
}
