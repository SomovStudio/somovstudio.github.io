Использование переменного числа аргументов params


public int MinVal(params int[] nums)
{
    if(nums.Length == 0){
        Console.WriteLine("");
        return 0;
    }
    
    int min = nums[0];
    for(int i = 1; i < nums.Length; i++)
    {
        if(nums[i] < min) min = nums[i];
    }
    return min;
}



Min objMin = new Min();
int min = objMin.MinVal(10, 5, 25, 11);
Console.WriteLine("Минимальное значение из 10, 5, 25, 11 будет " + min);

Использование обычных параметров вместе с параметром params


class Min
{
    public void Msg(string message, params int[] nums)
    {
        foreach(int i in nums)
        {
            Console.WriteLine(message + " " + i);
        }
    }
}


Min objMin = new Min();
objMin.Msg("Число: ", 10, 5, 25, 11);
