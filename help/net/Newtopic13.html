class TwoDShape
{
    private double _width;
    private double _height;
    public double Width {
        get{ return _width; }
        set{ _width = value; }
    }
    public double Height {
        get{ return _height; }
        set{ _height = value; }
    }
    public void ShowDim()
    {
        Console.WriteLine("Ширина: " + Width + " Высота: " + Height);
    }
}

class Triangle: TwoDShape
{
    public double Area()
    {
        return Width * Height / 2;
    }    
    public void ShowResult(int width, int height)
    {
        Width = width;
        Height = height;
        ShowDim();
        Console.WriteLine("Area: " + Area());
    }
}


Пример использования


Triangle tangle = new Triangle();
tangle.ShowResult(1024, 768);

Организация защищенного доступа


Защищенный член является открытым в пределах иерархии классов, но закрытым за пределами этой иерархии.

Защищенный член создается с помощью модификтора доступа protected.


class ClassA 
{
    protected int i, j;
    public void SetIJ(int a, int b)
    {
        i = a;
        j = b;
    }
    public void ShowIJ()
    {
        Console.WriteLine("CLASS A: i = " + i + " j = " + j);
    }
}

class ClassB: ClassA
{
    int k;
    public void InitK()
    {
        k = i * j;
    }
    public void ShowK()
    {
        Console.WriteLine("CLASS B: k = " + k);
    }
}


Пример использования


ClassB obj = new ClassB();
obj.SetIJ(10, 20);
obj.ShowIJ();
obj.InitK();
obj.ShowK();
