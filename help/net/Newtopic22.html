Наследование и сокрытие имен


class A
{
    public int i = 0;
}

class B : A 
{
    new int i;
    public B (int b)
    {
        i = b;
    }
    public void Show()
    {
        Console.WriteLine("CLASS B: Член i в производном классе: " + i);
    }
}


Пример использования


B obj = new B(5);
Console.WriteLine("CLASS A: Значение скрываемого члена i = " + obj.i);
obj.Show();

Применение ключевого слова base для доступа к скрытому имени


class A
{
    public int i = 0;
}

class B : A 
{
    new int i;
    public B (int b)
    {
        base.i = b;
        i = b;
    }
    public void Show()
    {
        Console.WriteLine("Член i в базовом классе: " + base.i);
        Console.WriteLine("Член i в производном классе: " + i);
    }
}
