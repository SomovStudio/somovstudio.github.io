Реализация интерфейсов


public interface ISeries
{
    int GetNext();
    void Reset();
    void SetStart(int x);
}

public class ByTwos: ISeries
{
    int start;
    int val;
    
    public ByTwos()
    {
        start = 0;
        val = 0;
    }
    public int GetNext()
    {
        val += 2;
        return val;
    }
    public void Reset()
    {
        val = start;
    }
    public void SetStart(int x)
    {
        start = x;
        val = start;
    }
}


Пример использования


ByTwos obj = new ByTwos();
for(int i = 0; i < 5; i++)
{
    Console.WriteLine("Следующее число равно " + obj.GetNext());
}

Console.WriteLine("Сбросить");
obj.Reset();

Console.WriteLine("Начать с числа 100");
obj.SetStart(100);
for(int i = 0; i < 5; i++)
{
    Console.WriteLine("Следующее число равно " + obj.GetNext());
}

Интерфейсные индексаторы


public interface ISeries
{
    int this[int index] {
        get;
    }
}

public class ByTwos : ISeries
{
    int val;
    public ByTwos()
    {
        val = 0;
    }
    public int this[int index]
    {
        get {
            val = 0;
            for(int i = 0; i < index; i++) val += 2;
            return val;
        }
    }
}


Пример использования


ByTwos obj = new ByTwos();
for(int i = 0; i < 5; i++)
{
    Console.WriteLine("Индексация: Следующее число равно " + obj[i]);
}

Наследование интерфейсов


public interface IA
{
    void Meth1();
}

public interface IB: IA
{
    void Meth2();
}

public class TestClass: IB
{
    public void Meth1()
    {
        Console.WriteLine("Вызов метода Meth1");
    }
    public void Meth2()
    {
        Console.WriteLine("Вызов метода Meth2");
    }
}


Пример использования


TestClass obj = new TestClass();
obj.Meth1();
obj.Meth2();

Явные реализации


interface IEven
{
    bool IsOdd(int x);
    bool IsEven(int x);
}

class TestClass: IEven
{
    bool IEven.IsOdd(int x)
    {
        if((x%2) != 0) return true;
        else return false;
    }
    public bool IsEven(int x)
    {
        IEven o = this;
        return !o.IsOdd(x);
    }
}


Пример использования


TestClass obj = new TestClass();
bool result;
result = obj.IsEven(4);
if(result) Console.WriteLine("4 - четное число.");
