Одиночные лямбда-выражения


delegate int Incr(int i);

public static class SimpleLambda
{
    public static void test()
    {
        Incr incr1 = (count) => count + 1;
        Incr incr2 = count => count + 2;
    
        int i = 0;
        while(i < 5) 
        {
            i = incr1(i);
            Console.WriteLine("Лямбда-выражение: значение i = " + i);
        }
    }
}

Блочные лямбда-выражения


delegate int IntOp(int i);

public static class SimpleLambda
{
    public static void test()
    {
        IntOp fact = n => {
            int r = 1;
            for(int i=1; i<=n; i++) r = i * r;
            return r;
        };
        Console.WriteLine("Лямбда-выражение: Факториал 3 равен " + fact(3));
    }
}
