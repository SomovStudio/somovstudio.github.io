Класс EventArgs


class MyEventArgs: EventArgs
{
    public int EventNum;
}

delegate void MyEventHandler(object source, MyEventArgs arg);

class MyEvent
{
    static int _count = 0;
    public event MyEventHandler SomeEvent;
    public void OnSomeEvent()
    {
        MyEventArgs arg = new MyEventArgs();
        if(SomeEvent != null)
        {
            arg.EventNum = _count++;
            SomeEvent(this, arg);
        }
    }
}

class ObjEX
{
    public void onEventX(object source, MyEventArgs arg) 
    {
        Console.WriteLine("Произошло событие X. Источник = " + source + " | Аргумент = " + arg.EventNum);
    }
}

class ObjEY
{
    public void onEventY(object source, MyEventArgs arg) 
    {
        Console.WriteLine("Произошло событие Y. Источник = " + source + " | Аргумент = " + arg.EventNum);
    }
}

static class EventDemo
{
    public static void test()
    {
        ObjEX objX = new ObjEX();
        ObjEY objY = new ObjEY();
        
        MyEvent evt = new MyEvent();
        evt.SomeEvent += objX.onEventX;
        evt.SomeEvent += objY.onEventY;
        
        evt.OnSomeEvent();
        evt.OnSomeEvent();
    }
}
