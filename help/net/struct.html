Структуры


struct Book 
{
    public string Author;
    public string Title;
    public Book(string author, string title)
    {
        Author = author;
        Title = title;
    }
}


Пример использования


Book book1;
book1.Author = "Автор Book 1";
book1.Title = "Заголовок Book 1";
Console.WriteLine(book1.Title + " " + book1.Author);

Book book2 = new Book();
book2.Author = "Автор Book 2";
book2.Title = "Заголовок Book 2";
Console.WriteLine(book2.Title + " " + book2.Author);

Book book3 = new Book("Автор Book 3", "Заголовок Book 3");
Console.WriteLine(book3.Title + " " + book3.Author);
