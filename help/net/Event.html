Простой пример события


delegate void MyEventHandler();

class MyEvent 
{
    public event MyEventHandler SomeEvent;
    public void OnSomeEvent()
    {
        if(SomeEvent != null) SomeEvent();
    }
}

public static class EventDemo
{
    public static void onEvent() 
    {
        Console.WriteLine("Произошло событие!");
    }
    public static void test()
    {
        MyEvent evt = new MyEvent();
        evt.SomeEvent += onEvent;
        evt.OnSomeEvent();
    }
}

delegate void MyEventHandler();

class MyEvent
{
    MyEventHandler[] evnt = new MyEventHandler[3];
    public event MyEventHandler SomeEvent {
        add {
            for(int i=0; i < evnt.Length; i++)
            {
                if(evnt[i] == null) {
                    evnt[i] = value;
                    break;
                }
            }
        }
        remove {
            for(int i=0; i < evnt.Length; i++)
            {
                if(evnt[i] == value) {
                    evnt[i] = null;
                    break;
                }
            }
        }
    }
    public void OnSomeEvent()
    {
        for(int i=0; i < evnt.Length; i++)
        {
            if(evnt[i] != null) evnt[i]();
        }
    }
}

Применение анонимных методов и лямбда-выражений вместе с событиями


delegate void MyEventHandler(int n);

class MyEvent 
{
    public event MyEventHandler SomeEvent;
    public void OnSomeEvent(int n)
    {
        if(SomeEvent != null) SomeEvent(n);
    }
}

static class EventDemo
{
    public static void test()
    {
        MyEvent evt = new MyEvent();
        evt.SomeEvent += (n) => Console.WriteLine("Произошло лямбда событие. Значение: " + n);
        evt.SomeEvent += delegate(int n) {
            Console.WriteLine("Произошло анонимное событие. Значение: " + n);
        };
        
        evt.OnSomeEvent(1);
        evt.OnSomeEvent(2);
    }
}
