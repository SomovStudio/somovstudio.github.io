Классы

Модификаторы доступа

public - публичный объект доступный везде

private - приватный объект доступный внутри класса

protected - применяется для случаев с наследованием

internal - служит для сборки (определяет доступность члена во всех файлах сборки и его недоступность за пределами сборки)


Простой класс


public class Building
{
    public int Floors;        // количество этажей
    public int Area;                // общая площадь здания
    public int Occupants;        // количество жильцов
    
    public void ShowResult()
    {
        Console.WriteLine(" " + Area / Occupants + " приходится на одного человека");
    }
}

public class BuildingDemo
{
    public void CreateBuild()
    {
        Building house = new Building();
        house.Occupants = 4;
        house.Area = 2500;
        house.Floors = 2;
        house.ShowResult();
    }
}


Конструкторы, диструктор и параметры класса


public class Building
{
    private int _floors;        // количество этажей
    private int _area;        // общая площадь здания
    private int _occupants;// количество жильцов
    
    public Building(int floors, int area, int occupants = 0) // конструктор
    {
        _floors = floors;
        _area = area;
        _occupants = occupants;
    }
    
    ~Building(){    // диструктор (сработает после завершения программы)
        Console.WriteLine("Диструктор класса! Floors =" + _floors);
        Console.WriteLine("Диструктор класса! Area = " + _area);
        Console.WriteLine("Диструктор класса! Occupants = " +_occupants);
    }
    
    public int Result()
    {
        return _area / _occupants;
    }
}

public class BuildingDemo
{
    public void CreateBuild()
    {
        Building house = new Building(4, 2500, 2);
        int result = house.Result();
        Console.WriteLine("RESULT: " + result);
    }
}


Ключевое слово this


public class Rect {
    public int Width;
    public int Height;
    public Rect(int w, int h) {
        this.Width = w;
        this.Height = h;
    }
    public int Area() {
        return this.Width * this.Height;
    }
}

public class UseRect {
    public static void MainTest() 
    {
        Rect r1 = new Rect(4, 5);
        Rect r2 = new Rect(7, 9);
        Console.WriteLine("Площадь прямоугольника r1: " + r1.Area());
        Console.WriteLine("Площадь прямоугольника r2: " + r2.Area());
    }
}
