Необязательные аргументы


public class MyClass
{
    public int Count;
    public string Str;
    
    public MyClass(string str = "INIT", int count = 0)
    {
        Count = count;
        Str = str;
    }
    
    public void setData(string str = "SET", int count = 100)
    {
        Count = count;
        Str = str;
    }
}


MyClass mclass = new MyClass();
Console.WriteLine("Инициализация класса: Str = " + mclass.Str + " Count = " + mclass.Count);
mclass.setData();
Console.WriteLine("Вызов метода: Str = " + mclass.Str + " Count = " + mclass.Count);
