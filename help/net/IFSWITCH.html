Управляющие операторы

Опертор if (условие)


int count = 0;
if(count < 0)
{
    Console.WriteLine("Количество меньше 0");
}
else if(count == 0)
{
    Console.WriteLine("Количество равно 0");
}
else
{
    Console.WriteLine("Количество больше 0");
}


Оператор switch (выбор)


string color = "red";
switch (color)
{
    case "red":
        Console.WriteLine("Цвет - красный");
        break;
    case "green":
        Console.WriteLine("Цвет - зелёный");
        break;
    case "blue":
        Console.WriteLine("Цвет - синий");
        break;
    default:
        Console.WriteLine("Не корректно выбран цвет!");
        break;
}


Оператор goto (переход)


int num = 0;
switch(num)
{
    case 0:
        Console.WriteLine("Число = 0");
        goto case 2;
    case 1:
        Console.WriteLine("Число = 1");
        break;
    case 2:
        Console.WriteLine("Число = 2");
        goto case 1;
    default:
        Console.WriteLine("Не корректно выбраное число!");
        break;
}
