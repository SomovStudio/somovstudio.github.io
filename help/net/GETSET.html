Свойства get и set


class SimpProp
{
    private int prop;
    public SimpProp() 
    { 
        prop = 0; 
    }
    
    public int MyProp
    {
        get {
            return prop;
        }
        set {
            prop = value;
        }
    }
}


Пример использования


SimpProp obj = new SimpProp();
obj.MyProp = 100;
Console.WriteLine("Свойство MyProp = " + obj.MyProp);

Применение инициализаторов объектов в свойствах


class MyClass
{
    public int Count { get; set; }
    public string Str { get; set; }
}


MyClass obj = new MyClass();
obj.Count = 250;
obj.Str = "Количество: ";
Console.WriteLine(obj.Str + obj.Count);
