ANY

Возвращает true, если любой из элементов входной последовательности отвечает условию.


class User
{
    public int id;
    public string name;
}


List<User> users = new List<User>();
users.Add(new User{id = 1, name = "Admin"});
users.Add(new User{id = 2, name = "User"});
users.Add(new User{id = 3, name = "Someone"});

bool any = users.Where(u => u.id == 2).Any();

bool any = users.Any(s => s.name.StartsWith("U"));
Console.WriteLine(any);
