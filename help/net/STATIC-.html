static - статичные классы и метода


public static class StaticClassDemo
{
    public static int Val = 100;
    public static int ValDiv2()
    {
        return Val / 2;
    }
    public static void Msg(string message)
    {
        Console.WriteLine(message);
    }
}



StaticClassDemo.Msg("Статический класс. Значение: " + StaticClassDemo.ValDiv2());

Статические конструкторы



public class StaticClassDemo
{
    public static int Val;
    
    static StaticClassDemo()
    {
        Val = 250;
    }
    
    public int ValDiv2()
    {
        return Val / 2;
    }
    public void Msg(string message)
    {
        Console.WriteLine(message);
    }
}



StaticClassDemo sClass = new StaticClassDemo();
sClass.Msg("Статический класс. Значение: " + sClass.ValDiv2());
