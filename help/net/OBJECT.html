Упаковка и распаковка


int x;
object obj;

x = 10;
obj = x; // упаковать значение переменной в объект

int y = (int)obj; // распаковать значение из обекта

Console.WriteLine("Упакованный объект obj = " + obj);
Console.WriteLine("Распакованный объект obj в переменную y = " + y);
