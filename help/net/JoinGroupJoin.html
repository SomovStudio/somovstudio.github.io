JOIN - соединение нескольких последовательностей


class User
{
    public int id;
    public string name;
}

class Action
{
    public int id;
    public DateTime date;
    public string description;
}


public static void Demo()
{
    List<User> users = new List<User>();
    users.Add(new User{id = 1, name = "Admin"});
    users.Add(new User{id = 2, name = "User"});
    
    User[] tableUsers = (User[]) users.ToArray();
                
    List<Action> actions = new List<Action>();
    actions.Add(new Action{ id = 2, date = new DateTime(2019, 1, 4), description = "Login" });
    actions.Add(new Action{ id = 2, date = new DateTime(2019, 1, 4), description = "LoginOut" });
    actions.Add(new Action{ id = 1, date = new DateTime(2019, 1, 5), description = "Login" });
    actions.Add(new Action{ id = 2, date = new DateTime(2019, 1, 5), description = "Login" });
    actions.Add(new Action{ id = 1, date = new DateTime(2019, 1, 5), description = "LoginOut" });
    
    Action[] tableActions = (Action[]) actions.ToArray();
    
    
    var values = tableUsers.Join(
        tableActions,
        e => e.id,
        o => o.id,
        (e, o) => new {
            id = e.id,    // последовательность inner 
            name = e.name,    // outerKeySelector
            date = o.date,    // innerKeySelector
            description = o.description    // resultSelector
        }
    );
    
    foreach(var value in values)
    {
        Console.WriteLine(value);
    }
}

GROUP JOIN - групповое соединение двух последовательностей на основе ключей


public static void DemoJoinGroup()
{
    List<User> users = new List<User>();
    users.Add(new User{id = 1, name = "Admin"});
    users.Add(new User{id = 2, name = "User"});
    
    User[] tableUsers = (User[]) users.ToArray();
                
    List<Action> actions = new List<Action>();
    actions.Add(new Action{ id = 2, date = new DateTime(2019, 1, 4), description = "Login" });
    actions.Add(new Action{ id = 2, date = new DateTime(2019, 1, 4), description = "LoginOut" });
    actions.Add(new Action{ id = 1, date = DateTime.Today, description = "Login" });
    actions.Add(new Action{ id = 2, date = DateTime.Today, description = "Login" });
    actions.Add(new Action{ id = 1, date = DateTime.Today, description = "LoginOut" });
    
    Action[] tableActions = (Action[]) actions.ToArray();
    
    
    var values = tableUsers.GroupJoin(
        tableActions,
        e => e.id,
        o => o.id,
        (e, os) => new {
            id = e.id,
            name = e.name,
            date = os.All(o => o.date == DateTime.Today)
        }
    );
    
    foreach(var value in values)
    {
        Console.WriteLine(value);
    }
}
