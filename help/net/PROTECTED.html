Организация защищенного доступа

Защищенный член является открытым в пределах иерархии классов, но закрытым за пределами этой иерархии.

Защищенный член создается с помощью модификтора доступа protected.


class ClassA 
{
    protected int i, j;
    public void SetIJ(int a, int b)
    {
        i = a;
        j = b;
    }
    public void ShowIJ()
    {
        Console.WriteLine("CLASS A: i = " + i + " j = " + j);
    }
}

class ClassB: ClassA
{
    int k;
    public void InitK()
    {
        k = i * j;
    }
    public void ShowK()
    {
        Console.WriteLine("CLASS B: k = " + k);
    }
}


Пример использования


ClassB obj = new ClassB();
obj.SetIJ(10, 20);
obj.ShowIJ();
obj.InitK();
obj.ShowK();
