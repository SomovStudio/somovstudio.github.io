Перечисления (enum)

C# ››

Перечисления


enum Fruits
{
    Apple,
    Banana,
    Pear,
    Cherry,
    Peach
}


Пример использования


public static class ShowFruits
{
    public static void Show()
    {
        Console.WriteLine("Яблоко : " + Fruits.Apple + " индекс: " + (int)Fruits.Apple);
        
        Fruits f;
        for(f = Fruits.Apple; f < Fruits.Peach; f++)
        {
            Console.WriteLine("Фрукт: " + f + " индекс: " + (int)f);
        }
    }
}
