ELEMENT AT

Возвращает элемент последовательности по его индексу


class User
{
    public int id;
    public string name;
}


List<User> users = new List<User>();
users.Add(new User{id = 1, name = "Admin"});
users.Add(new User{id = 2, name = "User"});
users.Add(new User{id = 3, name = "Someone"});

User user = users.ElementAt(2);
Console.WriteLine(user.id + " : " + user.name);

ELEMENT AT OR DEFAULT


List<User> users = new List<User>();
users.Add(new User{id = 1, name = "Admin"});
users.Add(new User{id = 2, name = "User"});
users.Add(new User{id = 3, name = "Someone"});

User user = users.ElementAtOrDefault(3);
Console.WriteLine(user == null ? "NULL" : user.name);
