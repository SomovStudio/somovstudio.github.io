Перегрузка операторов

Перегрузка операторов выполняется командой operator.


Перегрузка бинарных операторов + и -


public class Points
{
    private int x, y, z;
    
    public Points(){ x = y = z = 0; }
    public Points(int X, int Y, int Z){ 
        x = X; y = Y; z = Z;
    }
    
    public static Points operator + (Points op1, Points op2)
    {
        Points points = new Points();
        points.x = op1.x + op2.x;
        points.y = op1.y + op2.y;
        points.z = op1.z + op2.z;
        return points;
    }
    
    public static Points operator - (Points op1, Points op2)
    {
        Points points = new Points();
        points.x = op1.x - op2.x;
        points.y = op1.y - op2.y;
        points.z = op1.z - op2.z;
        return points;
    }
    
    public void Show(string msg)
    {
        Console.WriteLine(msg + ": x=" + x + " y=" + y + " z=" + z);
    }
}


Использоание перегруженных операторов


Points points1 = new Points(1, 2, 3);
Points points2 = new Points(10, 10, 10);
Points points3;

points1.Show("Координаты Points1");
points2.Show("Координаты Points2");

points3 = points1 + points2;
points3.Show("Результат сложения Points1 + Points2");

points3 = points2 - points1;
points3.Show("Результат вычитания Points2 - Points1");

Перегрузка унарных операторов ++ и --


public class Points
{
    private int x, y, z;
    
    public Points(){ x = y = z = 0; }
    public Points(int X, int Y, int Z){ 
        x = X;
        y = Y;
        z = Z;
    }
    
    public static Points operator ++ (Points op)
    {
        Points points = new Points();
        points.x = op.x + 1;
        points.y = op.y + 1;
        points.z = op.z + 1;
        return points;
    }
    
    public static Points operator -- (Points op)
    {
        Points points = new Points();
        points.x = op.x - 1;
        points.y = op.y - 1;
        points.z = op.z - 1;
        return points;
    }
    
    public void Show(string msg)
    {
        Console.WriteLine(msg + ": x=" + x + " y=" + y + " z=" + z);
    }
}

Перегрузка операторов отношения < и >


public class Points
{
    private int x, y, z;
    
    public Points(){ x = y = z = 0; }
    public Points(int X, int Y, int Z){ 
        x = X;
        y = Y;
        z = Z;
    }
    
    public static bool operator < (Points op1, Points op2)
    {
        if( (op1.x + op1.y + op1.z) < (op2.x + op2.y + op2.z) )
        {
            return true;
        }else{
            return false;
        }
    }
    
    public static bool operator > (Points op1, Points op2)
    {
        if( (op1.x + op1.y + op1.z) > (op2.x + op2.y + op2.z) )
        {
            return true;
        }else{
            return false;
        }
    }
    
    public void Show(string msg)
    {
        Console.WriteLine(msg + ": x=" + x + " y=" + y + " z=" + z);
    }
}


Использоание перегруженных операторов

Перегрузка операторов отношения true и false


public class Points
{
    private int x, y, z;
    
    public Points(){ x = y = z = 0; }
    public Points(int X, int Y, int Z){ 
        x = X;
        y = Y;
        z = Z;
    }
    
    public static bool operator true (Points op)
    {
        if( (op.x != 0) || (op.y != 0) || (op.z != 0) ) 
            return true;
        else
            return false;
    }
    
    public static bool operator false (Points op)
    {
        if( (op.x == 0) && (op.y == 0) && (op.z == 0) ) 
            return true;
        else
            return false;
    }
    
    public void Show(string msg)
    {
        Console.WriteLine(msg + ": x=" + x + " y=" + y + " z=" + z);
    }
}


Использоание перегруженных операторов


Points points = new Points();

else points.Show("Points = false");
