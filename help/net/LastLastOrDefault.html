LAST

Возвращает последний элемент последовательности


string[] users = {"Admin", "User", "sa", "John", "Jackson", "Bill"};
string user = users.Last();
Console.WriteLine(user);

LAST OR DEFAULT


string[] users = {"Admin", "User", "sa", "John", "Jackson", "Bill"};
string user = users.Take(0).LastOrDefault();
Console.WriteLine(user == null ? "NULL" : user);
