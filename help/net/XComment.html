Создание комментариев с помощью XComment


XElement xElem = new XElement("books", new XComment("This is a comment"));
Console.WriteLine(xElem);


или


XElement xElem = new XElement("books");
XComment xComm = new XComment("This is a comment");
xElem.Add(xComm);
Console.WriteLine(xElem);
