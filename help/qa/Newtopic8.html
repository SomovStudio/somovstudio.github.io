Последовательное выполнение нажатий с задержками и снятие скриншота

Скрипт: Последовательное выполнение нажатий с задержками и снятие скриншота

-----------------------------------------------------------------------------------------------

from com.android.monkeyrunner import MonkeyRunner, MonkeyDevice

device = MonkeyRunner.waitForConnection()

# Нажимает кнопку (Nexus7 1280х800)

device.touch(640, 650, 'DOWN_AND_UP')

print("CLICK - 1");

MonkeyRunner.sleep(15.0)                        # задержка

device.touch(640, 650, 'DOWN_AND_UP')

print("CLICK - 2");

MonkeyRunner.sleep(5.0)                        # задержка

device.touch(640, 650, 'DOWN_AND_UP')

print("CLICK - 3");

result = device.takeSnapshot()

result.writeToFile('C:/logs/builds/shot.png','png')

print("SCREENSHOT - SAVE");
