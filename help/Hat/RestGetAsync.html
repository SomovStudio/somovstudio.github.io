RestGetAsync

Описание: метод выполняет Get Rest запрос и получает результат в формате json

Синтаксис: RestGetAsync(string url, string charset = "UTF-8")


Пример:

string result = await tester.RestGetAsync("https://jsonplaceholder.typicode.com/posts/1/");

tester.ConsoleMsg(result);


Данный метод использует стандартный подход:

using System.Net;

using System.Net.Http;

using System.Net.Http.Headers;


Uri uri = new Uri(url);

HttpClient client = new HttpClient();

client.BaseAddress = uri;

client.DefaultRequestHeaders.Clear();

client.DefaultRequestHeaders.Accept.Clear();

client.DefaultRequestHeaders.Accept.Add(new MediaTypeWithQualityHeaderValue("application/json"));

client.DefaultRequestHeaders.Add("charset", "UTF-8");

client.DefaultRequestHeaders.Add("User-Agent", userAgent);

HttpResponseMessage response = await client.GetAsync(url);

if (response.IsSuccessStatusCode)

{

       return await response.Content.ReadAsStringAsync();

}
