Базовая авторизация


Бывают такие случаи когда страница закрыта базовой авторизацией



Для работы с базовой авторизацией есть два метода: 

  • BrowserBasicAuthenticationAsync - выполняет авторизацию до выполнения шагов автотеста
  • GoToUrlBaseAuthAsync - выполняет переход на страницу при этом в адресной URL прописаны логин и пароль


Пример использования метода BrowserBasicAuthenticationAsync

using System.Reflection;

using HatFramework;


namespace Hat

{

    public class ExampleTest5

    {

        Tester tester;


        public async void Main(Form browserWindow)

        {

            tester = new Tester(browserWindow);


            await setUp();

            await test();

            await tearDown();

        }


        public async Task setUp()

        {

            await tester.BrowserFullScreenAsync();

            await tester.BrowserBasicAuthenticationAsync("login", "pass"); // базовая авторизация

        }


        public async Task test()

        {

            await tester.TestBeginAsync();

            await tester.GoToUrlAsync("https://test.site.com", 5);

            await tester.TestEndAsync();

        }


        public async Task tearDown()

        {

            // await tester.BrowserCloseAsync();

        }

    }

}
